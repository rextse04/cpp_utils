828 if (!dptr) {
829 control = nullptr;
830 return;
831 }
832 ReAlloc alloc_ = alloc;
833 auto control_ptr = ReAllocTraits::allocate(alloc_, 1);
834 control = std::to_address(control_ptr);
835 ReAllocTraits::construct(alloc_, std::to_address(control_ptr),
836 dptr.template rebind_deleter<BaseDeleter>(std::forward<D>(deleter)), std::move(alloc_), control_ptr);
837 }

841 constexpr shared_dptr_deleter(shared_dptr_control_base* control, const auto& expired_handler) {
842 if (control) {
843 long shared_count = control->shared_count.load(std::memory_order::acquire);
844 while (shared_count) {
845 if (control->shared_count.compare_exchange_weak(shared_count, shared_count + 1,
846 std::memory_order::acq_rel, std::memory_order::acquire)) {
847 this->control = control;
848 return;
849 }
850 }
851 }
852 expired_handler();
853 }
