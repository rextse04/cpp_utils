247 constexpr void build() {
248 level_offsets_[levels_ = 1] = keys().size();
249 size_type prev = 0;
250 while (level_offsets_[levels_] - level_offsets_[levels_ - 1] > 1) {
251 const size_type prev_end = nodes().size();
252 while (prev < prev_end - 1) {
253 m_nodes().push_back(sum_(nodes()[prev], nodes()[prev + 1]));
254 prev += 2;
255 }
256 if (prev != prev_end) {
257 m_nodes().push_back(sum_(nodes()[prev], identity_()));
258 ++prev;
259 }
260 level_offsets_[++levels_] = nodes().size();
261 }

622 template <bool LowerBound>
623 constexpr size_type search(const auto& key) const {
624 const auto& k = make_search_key<segment_tree>(key);
625 if constexpr (LowerBound) {
626 return ranges::lower_bound(keys().begin(), keys().begin() + size(), k, comp_) - keys().begin();
627 } else {
628 return ranges::upper_bound(keys().begin(), keys().begin() + size(), k, comp_) - keys().begin();
629 }

631 template <bool Throw>
632 constexpr size_type exact_search(const auto& key) const {
633 const size_type found = search<true>(key);
634 if (found == size() || comp_(key, keys()[found]) || comp_(keys()[found], key)) {
635 if constexpr (Throw) throw std::out_of_range("key not found in segment_tree.");
636 else return size();
637 }
638 return found;
639 }

740 template <typename U>
741 constexpr void do_assign(size_type node, U&& mapped) {
742 m_nodes()[node] = std::forward<U>(mapped);
743 for (int level = 1; level < levels_; ++level) {
744 const bool odd = node % 2;
745 const size_type prev = level_offsets_[level - 1] + node;
746 node /= 2;
747 const size_type current = level_offsets_[level] + node;
748 if (odd) {
749 m_nodes()[current] = sum_(m_nodes()[prev], m_nodes()[prev - 1]);
750 } else {
751 if (prev + 1 == level_offsets_[level]) {
752 m_nodes()[current] = sum_(m_nodes()[prev], identity_());
753 } else {
754 m_nodes()[current] = sum_(m_nodes()[prev], m_nodes()[prev + 1]);
755 }
756 }
757 }

820 struct node_info {
821 int level, left_level, right_level;
822 size_type level_idx, left_idx, right_idx, begin, end, div_node;
823 };
824 private:
825 constexpr auto do_traverse(this auto&& self, const auto& traverser, const auto& strategy,
826 int level, size_type level_idx, size_type begin, size_type end) {
827 using enum interval_endpoint_type;
828 node_info info;
829 info.level = level;
830 info.level_idx = level_idx;
831 info.begin = begin;
832 info.end = end;
833 info.div_node = begin + (std::bit_ceil(end - begin) >> 1);
834 info.left_level = level - 1;
835 info.left_idx = level_idx * 2;
836 info.right_level = std::bit_width(end - info.div_node - 1);
837 info.right_idx = ((level_idx * 2 + 1) << (level - info.right_level)) >> 1;
838 const node_info* info_ptr = std::addressof(info);
839 const segment_tree_traverse_direction dir = traverser(self, info);
840 if (dir == stop) return strategy(self, info_ptr, nullptr, nullptr);
841 if (level == 0) return strategy(self, nullptr, nullptr, nullptr);
842 if constexpr (requires { strategy.prepare(self, *info_ptr); }) strategy.prepare(self, *info_ptr);
843 const auto collect_left = [&] {
844 return self.do_traverse(traverser, strategy, info.left_level, info.left_idx, info.begin, info.div_node);
845 };
846 const auto collect_right = [&] {
847 return self.do_traverse(traverser, strategy, info.right_level, info.right_idx, info.div_node, info.end);
848 };
849 if (dir == left) {
850 const auto& left = collect_left();
851 return strategy(self, info_ptr, std::addressof(left), nullptr);
852 }
853 if (dir == right) {
854 const auto& right = collect_right();
855 return strategy(self, info_ptr, nullptr, std::addressof(right));
856 }
857 if (dir == both) {
858 const auto& left = collect_left();
859 const auto& right = collect_right();
860 return strategy(self, info_ptr, std::addressof(left), std::addressof(right));
861 }
862 std::unreachable();
863 }
864 template <typename K1, typename K2>
865 struct key_traverser {
866 interval<K1, K2> itv_;
867
868 constexpr auto operator()(const segment_tree& base, const node_info& info) const
869 -> segment_tree_traverse_direction {
870 using enum interval_endpoint_type;
871 const interval<const Key&, const Key&> root_itv(
872 {base.keys()[info.begin], closed}, {base.keys()[info.end - 1], closed});
873 if (itv_.superset_of(root_itv, base.comp_)) return stop;
874 const interval_endpoint<const Key&> div_right(base.keys()[info.div_node], open);
875 if (itv_.right.compare(div_right, true, base.comp_) <= 0) return left;
876 const interval_endpoint<const Key&> div_left(base.keys()[info.div_node], closed);
877 if (itv_.left.compare(div_left, false, base.comp_) <= 0) return right;
878 return both;
879 }
880 };
881 struct node_traverser {
882 size_type begin_, end_;
883
884 constexpr auto operator()(const segment_tree&, const node_info& info) const noexcept
885 -> segment_tree_traverse_direction {
886 if (begin_ <= info.begin && end_ >= info.end) return stop;
887 if (end_ <= info.div_node) return left;
888 if (begin_ >= info.div_node) return right;
889 return both;
890 }
891 };
892 struct aggregate_strategy {
893 template <typename Info, typename L, typename R>
894 constexpr mapped_type operator()(const segment_tree& base, Info info, L left, R right) const noexcept {
895 if constexpr (std::is_null_pointer_v<L> && std::is_null_pointer_v<R>) {
896 if constexpr (std::is_null_pointer_v<Info>) return base.identity_();
897 else return base.nodes()[base.level_offsets_[info->level] + info->level_idx];
898 }
899 else if constexpr (std::is_null_pointer_v<R>) return *left;
900 else if constexpr (std::is_null_pointer_v<L>) return *right;
901 else return base.sum_(*left, *right);
902 }
903 };
904 template <typename Mul>
905 struct aggregate_with_update_strategy {
906 private:
907 update_type& update_;
908 const Mul& multiplies_;
909 public:
910 constexpr aggregate_with_update_strategy(update_type& update, const Mul& multiplies) noexcept :
911 update_(update), multiplies_(multiplies) {}
912 constexpr void prepare(const segment_tree& base, const node_info& info) const noexcept {
913 const size_type node = base.level_offsets_[info.level] + info.level_idx,
914 left = base.level_offsets_[info.left_level] + info.left_idx,
915 right = base.level_offsets_[info.right_level] + info.right_idx;
916 mapped_type& lazy = update_[base.nodes().size() + node];
917 mapped_type& lazy_left = update_[base.nodes().size() + left];
918 mapped_type& lazy_right = update_[base.nodes().size() + right];
919 update_[left] = base.sum_(update_[left], multiplies_(lazy, 1 << info.left_level));
920 update_[right] = base.sum_(update_[right], multiplies_(lazy, 1 << info.right_level));
921 lazy_left = base.sum_(lazy_left, lazy);
922 lazy_right = base.sum_(lazy_right, lazy);
923 lazy = base.identity_();
924 }
925 template <typename Info, typename L, typename R>
926 constexpr mapped_type operator()(const segment_tree& base, Info info, L left, R right) const noexcept {
927 if constexpr (std::is_null_pointer_v<L> && std::is_null_pointer_v<R>) {
928 if constexpr (std::is_null_pointer_v<Info>) return base.identity_();
929 else {
930 const size_type node = base.level_offsets_[info->level] + info->level_idx;
931 return base.sum_(base.nodes()[node], update_[node]);
932 }
933 }
934 else if constexpr (std::is_null_pointer_v<L>) return *right;
935 else if constexpr (std::is_null_pointer_v<R>) return *left;
936 else return base.sum_(*left, *right);
937 }
938 };
939 template <typename Mul>
940 struct update_strategy {
941 private:
942 update_type& update_;
943 const mapped_type& diff_;
944 const Mul& multiplies_;
945 public:
946 constexpr update_strategy(update_type& update, const mapped_type& diff, const Mul& multiplies) noexcept :
947 update_(update), diff_(diff), multiplies_(multiplies) {}
948 template <typename Info, typename L, typename R>
949 constexpr int operator()(const segment_tree& base, Info info, L, R) const noexcept {
950 if constexpr (std::is_null_pointer_v<Info>) return 0;
951 else {
952 const size_type node = base.level_offsets_[info->level] + info->level_idx;
953 if constexpr (std::is_null_pointer_v<L> && std::is_null_pointer_v<R>) {
954 mapped_type& update = update_[node];
955 mapped_type& lazy = update_[base.nodes().size() + node];
956 update = base.sum_(update, multiplies_(diff_, 1 << info->level));
957 lazy = base.sum_(lazy, diff_);
958 } else {
959 const size_type left = base.level_offsets_[info->left_level] + info->left_idx,
960 right = base.level_offsets_[info->right_level] + info->right_idx;
961 update_[node] = base.sum_(update_[left], update_[right]);
962 }
963 return 0;
964 }
965 }
966 };
